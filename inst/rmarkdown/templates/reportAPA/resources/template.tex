% Inspired in the following post https://tex.stackexchange.com/questions/195491/ifthenelse-equal-string-comparison-fails
\newcommand{\authorsLanguage}[1]{
  \ifthenelse{\equal{#1}{spanish}}
  {Autores: \\[\baselineskip]}%TRUE
  {Authors: \\[\baselineskip]}%FALSE
}

\newcommand*{\titleGP}{\begingroup % Create the command for including the title page in the document
\centering % Center all text
\vspace*{\baselineskip} % White space at the top of the page

\rule{\textwidth}{1.6pt}\vspace*{-\baselineskip}\vspace*{2pt} % Thick horizontal line
\rule{\textwidth}{0.4pt}\\[\baselineskip] % Thin horizontal line

{\LARGE $title$}\\[0.2\baselineskip] % Title

\rule{\textwidth}{0.4pt}\vspace*{-\baselineskip}\vspace{3.2pt} % Thin horizontal line
\rule{\textwidth}{1.6pt}\\[\baselineskip] % Thick horizontal line

\scshape % Small caps
$subtitle$ \\[\baselineskip] % Tagline(s) or further description
$date$\par % Location and year

\vspace*{2\baselineskip} % Whitespace between location/year and editors


\authorsLanguage{$language$} %See command definition above [DM]


$if(authors)$
    $for(authors)$
    \Large{$authors.name$}\\
    \small{$authors.affiliation$}\\
    $endfor$
$endif$


\vfill
\includegraphics[width=2cm]{logo.png} \\% Publisher logo
{\scshape $year$} \\[0.3\baselineskip] % Year published
{\large $publisher$}\par % Publisher

\rule{\widthof{$publisher$}}{0.4pt}\vspace*{-\baselineskip}\vspace{3.2pt}\\ % Thin horizontal line
\rule{\widthof{$publisher$}}{1.6pt}\\[\baselineskip] % Thick horizontal line

\endgroup}
